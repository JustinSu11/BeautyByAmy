> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Replace the existing dark-theme UI with the v0-designed light/cream beauty aesthetic while preserving all Square API and Stripe business logic.

## Square Service ظْ Booking Type Mapping

The existing `useServices()` hook returns:
```ts
{ id, name, description, variationId, price /* cents */, currency, rawDuration /* ms */, formattedDuration, teamMemberIds }
```

The v0 BookingContext expects:
```ts
{ id, name, category, duration /* minutes */, price /* dollars */, description? }
```

**Strategy**: Define a unified `BookingService` type in `lib/booking-context.tsx` that matches the Square shape but adds `category`. The adapter lives in `ServiceList` when fetching from Square. For now, category is derived from service name keywords.

**Step 2: Write new BookingContext**

The new context must:
1. Use Square's `Service` type from `useServices()` (has `variationId`, `price` in cents, `rawDuration` in ms)
2. Support `selectedServices: Service[]` (multi-select)
3. Expose `toggleService`, `isServiceSelected`, `totalPrice` (in cents), `totalDuration` (in ms)
4. Keep `selectedDate`, `selectedTime`, step machine (`'services' | 'datetime' | 'summary'`)

```tsx
'use client'

import { createContext, useContext, useState, useCallback, type ReactNode } from 'react'

// Square service shape from useServices() hook
export interface BookingService {
  id: string
  name: string
  description?: string
  variationId: string
  price: number       // in cents
  currency: string
  rawDuration: number // in milliseconds
  formattedDuration: string
  teamMemberIds: string[]
  category?: 'eyelashes' | 'brows' | 'permanent-makeup'
}

export type BookingStep = 'services' | 'datetime' | 'summary'

interface BookingState {
  step: BookingStep
  selectedServices: BookingService[]
  selectedDate: Date | null
  selectedTime: string | null
}

interface BookingContextValue extends BookingState {
  setStep: (step: BookingStep) => void
  toggleService: (service: BookingService) => void
  isServiceSelected: (serviceId: string) => boolean
  setSelectedDate: (date: Date | null) => void
  setSelectedTime: (time: string | null) => void
  totalPriceCents: number
  totalDurationMs: number
  canProceed: boolean
  reset: () => void
}

const BookingContext = createContext<BookingContextValue | null>(null)

export function BookingProvider({ children }: { children: ReactNode }) {
  const [step, setStep] = useState<BookingStep>('services')
  const [selectedServices, setSelectedServices] = useState<BookingService[]>([])
  const [selectedDate, setSelectedDate] = useState<Date | null>(null)
  const [selectedTime, setSelectedTime] = useState<string | null>(null)

  const toggleService = useCallback((service: BookingService) => {
    setSelectedServices((prev) => {
      const exists = prev.find((s) => s.id === service.id)
      if (exists) return prev.filter((s) => s.id !== service.id)
      return [...prev, service]
    })
  }, [])

  const isServiceSelected = useCallback(
    (serviceId: string) => selectedServices.some((s) => s.id === serviceId),
    [selectedServices]
  )

  const totalPriceCents = selectedServices.reduce((sum, s) => sum + s.price, 0)
  const totalDurationMs = selectedServices.reduce((sum, s) => sum + s.rawDuration, 0)

  const canProceed =
    (step === 'services' && selectedServices.length > 0) ||
    (step === 'datetime' && selectedDate !== null && selectedTime !== null) ||
    step === 'summary'

  const reset = useCallback(() => {
    setStep('services')
    setSelectedServices([])
    setSelectedDate(null)
    setSelectedTime(null)
  }, [])

  return (
    <BookingContext.Provider
      value={{
        step,
        setStep,
        selectedServices,
        toggleService,
        isServiceSelected,
        selectedDate,
        setSelectedDate,
        selectedTime,
        setSelectedTime,
        totalPriceCents,
        totalDurationMs,
        canProceed,
        reset,
      }}
    >
      {children}
    </BookingContext.Provider>
  )
}

export function useBooking() {
  const ctx = useContext(BookingContext)
  if (!ctx) throw new Error('useBooking must be used within BookingProvider')
  return ctx
}
```

Replace the static data imports and component body:

```tsx
'use client'

import { useState } from 'react'
import { useServices } from '@/hooks/useServices'
import type { BookingService } from '@/context/BookingContext'
import { ServiceCard } from './service-card'
import { Eye, PenLine, Sparkles, Loader2 } from 'lucide-react'
import { cn } from '@/lib/utils'

type Category = 'eyelashes' | 'brows' | 'permanent-makeup'

const categories: { id: Category; label: string }[] = [
  { id: 'eyelashes', label: 'Eyelashes' },
  { id: 'brows', label: 'Brows' },
  { id: 'permanent-makeup', label: 'Permanent Makeup' },
]

const categoryIcons = {
  eyelashes: Eye,
  brows: PenLine,
  'permanent-makeup': Sparkles,
}

// Keyword-based category detection from Square service names
function detectCategory(name: string): Category {
  const lower = name.toLowerCase()
  if (lower.includes('lash') || lower.includes('eyelash')) return 'eyelashes'
  if (lower.includes('micro') || lower.includes('powder') || lower.includes('combo') || lower.includes('lip') || lower.includes('pmu')) return 'permanent-makeup'
  return 'brows'
}

export function ServiceList() {
  const [activeCategory, setActiveCategory] = useState<Category>('eyelashes')
  const { services: rawServices, loading, error } = useServices()

  const services: BookingService[] = (rawServices ?? []).map((s) => ({
    ...s,
    category: detectCategory(s.name),
  }))

  const filtered = services.filter((s) => s.category === activeCategory)

  if (loading) {
    return (
      <div className="flex items-center justify-center py-12">
        <Loader2 className="h-6 w-6 animate-spin text-muted-foreground" />
      </div>
    )
  }

  if (error) {
    return (
      <div className="rounded-lg border border-destructive/30 bg-destructive/10 p-4 text-sm text-destructive">
        Unable to load services. Please try again.
      </div>
    )
  }

  return (
    <div>
      <div className="mb-6 flex gap-2 overflow-x-auto pb-1 scrollbar-none" role="tablist">
        {categories.map((cat) => {
          const Icon = categoryIcons[cat.id]
          const isActive = activeCategory === cat.id
          const count = services.filter((s) => s.category === cat.id).length
          return (
            <button
              key={cat.id}
              role="tab"
              aria-selected={isActive}
              onClick={() => setActiveCategory(cat.id)}
              className={cn(
                'flex shrink-0 items-center gap-2 rounded-full px-4 py-2.5 text-sm font-medium transition-all',
                isActive
                  ? 'bg-charcoal text-primary-foreground shadow-sm'
                  : 'bg-secondary text-secondary-foreground hover:bg-muted'
              )}
            >
              <Icon className="h-4 w-4" />
              {cat.label}
              <span className={cn(
                'rounded-full px-1.5 py-0.5 text-xs',
                isActive ? 'bg-white/20 text-primary-foreground' : 'bg-background text-muted-foreground'
              )}>
                {count}
              </span>
            </button>
          )
        })}
      </div>

      <div className="flex flex-col gap-3" role="tabpanel">
        {filtered.length === 0 ? (
          <p className="py-8 text-center text-sm text-muted-foreground">No services in this category.</p>
        ) : (
          filtered.map((service) => (
            <ServiceCard key={service.id} service={service} />
          ))
        )}
      </div>
    </div>
  )
}
```
